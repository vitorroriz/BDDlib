635  void rehash(size_t num_buckets)
636  {
637  num_buckets = std::max(num_buckets, static_cast<size_t>(std::ceil(num_elements / static_cast<double>(_max_load_factor))));
638  if (num_buckets == 0)
639  {
640  reset_to_empty_state();
641  return;
642  }
643  auto new_prime_index = hash_policy.next_size_over(num_buckets);
644  if (num_buckets == bucket_count())
645  return;
646  int8_t new_max_lookups = compute_max_lookups(num_buckets);
647  EntryPointer new_buckets(AllocatorTraits::allocate(*this, num_buckets + new_max_lookups));
648  for (EntryPointer it = new_buckets, real_end = it + static_cast<ptrdiff_t>(num_buckets + new_max_lookups - 1); it != real_end; ++it)
649  {
650  it->distance_from_desired = -1;
651  }
652  new_buckets[num_buckets + new_max_lookups - 1].distance_from_desired = Entry::special_end_value;
653  std::swap(entries, new_buckets);
654  std::swap(num_slots_minus_one, num_buckets);
655  --num_slots_minus_one;
656  hash_policy.commit(new_prime_index);
657  int8_t old_max_lookups = max_lookups;
658  max_lookups = new_max_lookups;
659  num_elements = 0;
660  for (EntryPointer it = new_buckets, end = it + static_cast<ptrdiff_t>(num_buckets + old_max_lookups); it != end; ++it)
661  {
662  if (it->has_value())
663  {
664  emplace(std::move(it->value));
665  it->destroy_value();
666  }
667  }
668  deallocate_data(new_buckets, num_buckets, old_max_lookups);
669  }

834  template<typename Key, typename... Args>
835  SKA_NOINLINE(std::pair<iterator, bool>) emplace_new_key(int8_t distance_from_desired, EntryPointer current_entry, Key && key, Args &&... args)
836  {
837  using std::swap;
838  if (num_slots_minus_one == 0 || distance_from_desired == max_lookups || static_cast<double>(num_elements + 1) / static_cast<double>(bucket_count()) > _max_load_factor)
839  {
840  grow();
841  return emplace(std::forward<Key>(key), std::forward<Args>(args)...);
842  }
843  else if (current_entry->is_empty())
844  {
845  current_entry->emplace(distance_from_desired, std::forward<Key>(key), std::forward<Args>(args)...);
846  ++num_elements;
847  return { { current_entry }, true };
848  }
849  value_type to_insert(std::forward<Key>(key), std::forward<Args>(args)...);
850  swap(distance_from_desired, current_entry->distance_from_desired);
851  swap(to_insert, current_entry->value);
852  iterator result = { current_entry };
853  for (++distance_from_desired, ++current_entry;; ++current_entry)
854  {
855  if (current_entry->is_empty())
856  {
857  current_entry->emplace(distance_from_desired, std::move(to_insert));
858  ++num_elements;
859  return { result, true };
860  }
861  else if (current_entry->distance_from_desired < distance_from_desired)
862  {
863  swap(distance_from_desired, current_entry->distance_from_desired);
864  swap(to_insert, current_entry->value);
865  ++distance_from_desired;
866  }
867  else
868  {
869  ++distance_from_desired;
870  if (distance_from_desired == max_lookups)
871  {
872  swap(to_insert, result.current->value);
873  grow();
874  return emplace(std::move(to_insert));
875  }
876  }
877  }
