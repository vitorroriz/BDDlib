74  /* Primary inputs and outputs */

76  primary_output = string("OUTPUT") [bind(&bench_format::bench_node_type::gate_type, _val)=_1] >> rhs_id [bind(&bench_format::bench_node_type::label, _val)=_1] >> +eol;
